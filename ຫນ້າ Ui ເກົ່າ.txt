  // --- ระบบจัดการข้อมูลออฟไลน์ (Offline Mode) ---
  Future<void> saveOrderOffline(Map<String, dynamic> orderData) async {
    final prefs = await SharedPreferences.getInstance();
    List<String> offlineOrders = prefs.getStringList('offline_orders') ?? [];
    offlineOrders.add(json.encode(orderData));
    await prefs.setStringList('offline_orders', offlineOrders);
  }

  Future<void> syncOfflineOrders() async {
    final prefs = await SharedPreferences.getInstance();
    List<String> offlineOrders = prefs.getStringList('offline_orders') ?? [];
    if (offlineOrders.isEmpty) return;
    
    List<String> remainingOrders = [];
    for (String orderStr in offlineOrders) {
      try {
        var response = await http.post(
          Uri.parse('$serverUrl/orders'), 
          headers: {"Content-Type": "application/json"}, 
          body: orderStr
        ).timeout(const Duration(seconds: 10)); // ปรับเวลาลงมา ไม่ให้แอปค้างนาน
        
        if (response.statusCode != 200) remainingOrders.add(orderStr);
      } catch (e) { 
        remainingOrders.add(orderStr); 
      }
    }
    await prefs.setStringList('offline_orders', remainingOrders);
  }

  double safeParse(dynamic value) {
    if (value == null) return 0.0;
    if (value is num) return value.toDouble();
    return double.tryParse(value.toString().replaceAll(',', '')) ?? 0.0;
  }

  Future<void> fetchProducts() async {
    setState(() { isLoading = true; errorMessage = ""; });
    final prefs = await SharedPreferences.getInstance(); 
    
    try {
      syncOfflineOrders(); 
      var response = await http.get(Uri.parse('$serverUrl/products')).timeout(const Duration(seconds: 5));
      
      if (response.statusCode == 200) {
        String responseBody = utf8.decode(response.bodyBytes);
        setState(() { 
          products = json.decode(responseBody); 
          isLoading = false; 
        });
        await prefs.setString('cached_products', responseBody); 
      } else {
        throw Exception("Server Error"); 
      }
    } catch (e) { 
      // 🔌 โหมดออฟไลน์: ดึงของเก่ามาโชว์แบบเนียนๆ ไม่ต้องแจ้งเตือนให้ตกใจ
      String? cachedData = prefs.getString('cached_products');
      
      if (cachedData != null && cachedData.isNotEmpty) {
        setState(() { 
          products = json.decode(cachedData); 
          isLoading = false; 
        });
      } else {
        // ข้อมูลตั้งต้นกรณีเพิ่งลงแอปใหม่และยังไม่เคยต่อเน็ตเลย
        String dummyData = '''
        [
          {"id": 1, "name": "Slow Bar Moka Pot", "price": 25000, "category": "Coffee", "stock": 99},
          {"id": 2, "name": "Aram Espresso", "price": 30000, "category": "Coffee", "stock": 99},
          {"id": 3, "name": "Hand Drip Coffee", "price": 35000, "category": "Coffee", "stock": 99},
          {"id": 4, "name": "Matcha Latte", "price": 28000, "category": "Tea", "stock": 99}
        ]
        ''';
        setState(() { 
          products = json.decode(dummyData); 
          isLoading = false; 
        });
      }
    }
  }

  void confirmRefund(int orderId) {
    showDialog(
      context: context,

            style: ElevatedButton.styleFrom(backgroundColor: Colors.red),

          )
        ]

  }

  Future<void> processRefund(int orderId) async {
    try {
      var response = await http.delete(Uri.parse('$serverUrl/orders/$orderId')).timeout(const Duration(seconds: 10));
      if (response.statusCode == 200) { 
        showSuccessDialog("✅ ຍົກເລີກບິນສຳເລັດ! (ຄືນ Stock ແລ້ວ)"); 
        fetchProducts(); 
      } else { 
        showSuccessDialog("ลบไม่ได้: ${response.statusCode}"); 
      }
    } catch (e) { 
      showSuccessDialog("ເກີດຂໍ້ຜິດພາດ: ບໍ່ສາມາດລຶບບິນໃນໂໝດອອຟລາຍໄດ້"); 
    }
  }

  IconData getCategoryIcon(String category) {
    switch (category) {
      case 'Coffee': return Icons.coffee;
      case 'Tea': return Icons.emoji_food_beverage;
      case 'Dessert': return Icons.cake;
      case 'Other': return Icons.local_drink;
      default: return Icons.fastfood;
    }
  }

  // --- ระบบจัดการตะกร้าสินค้า ---
  void addToCart(var product, String type, String sweet, String note, double finalPrice) {
    setState(() {
      var existingIndex = cart.indexWhere((item) => item['id'] == product['id'] && item['type'] == type && item['sweet'] == sweet);
      if (existingIndex != -1) { 
        cart[existingIndex]['qty']++; 
      } else { 
        cart.add({
          "id": product['id'], 
          "name": product['name'], 
          "price": finalPrice, 
          "qty": 1, 
          "type": type, 
          "sweet": sweet, 
          "note": note, 
          "category": product['category']
        }); 
      }
    });
  }

  void decreaseQty(int index) { 
    setState(() { 
      if (cart[index]['qty'] > 1) { 
        cart[index]['qty']--; 
      } else { 
        cart.removeAt(index); 
      } 
    }); 
  }
  
  void increaseQty(int index) { setState(() { cart[index]['qty']++; }); }
  void removeFromCart(int index) { setState(() { cart.removeAt(index); }); }

  Future<void> placeOrder(String paymentMethod) async {
    if (cart.isEmpty) return;
    setState(() => isLoading = true);
    Navigator.pop(context); 
    Navigator.pop(context); 
    
    var orderData = { 
      "total_amount": cart.fold(0.0, (sum, item) => sum + (item['price'] * item['qty'])), 
      "payment_method": paymentMethod, 
      "employee_id": 1, 
      "items": cart.map((i) => { 
        "product_id": i['id'], 
        "quantity": i['qty'], 
        "price_at_sale": i['price'], 
        "sweetness": i['sweet'] ?? "", 
        "item_type": i['type'] ?? "", 
        "note": i['note'] ?? "" 
      }).toList() 
    };
    
    try {
      var response = await http.post(
        Uri.parse('$serverUrl/orders'), 
        headers: {"Content-Type": "application/json"}, 
        body: json.encode(orderData)
      ).timeout(const Duration(seconds: 8));
      
      if (response.statusCode == 200) { 
        showSuccessDialog("ຊຳລະເງິນສຳເລັດ ($paymentMethod) ✅"); 
      } else { 
        showSuccessDialog("❌ ขายບໍ່ໄດ້: Error ${response.statusCode}"); 
      }
    } catch (e) {
      await saveOrderOffline(orderData);
      showSuccessDialog("💾 ບັນທຶກບິນລົງເຄື່ອງແລ້ວ (ອອຟລາຍ)");
    } finally {
      setState(() { cart.clear(); isLoading = false; });
    }
  }
